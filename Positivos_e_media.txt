import java.util.ArrayList;
import java.util.Scanner;
import java.text.DecimalFormat;

public class Main {

            public static void main(String[] args) {
                Scanner scan = new Scanner(System.in);
                DecimalFormat dec = new DecimalFormat("#.#");

                ArrayList<Float> numero = new ArrayList<>();
                ArrayList<Float> positive = new ArrayList<>();


                for(int i = 0; i < 6; i++){
                    Float x = scan.nextFloat();
                    numero.add(x);
                }
                for(int i = 0; i < 6; i++){
                    if(numero.get(i) > 0){
                        positive.add(numero.get(i));
                    }
                }

                int tamanho = positive.size();
                Float media = 0f;

                for(int i = 0; i < tamanho; i++){
                    media = media + positive.get(i);
                }

                media = media/positive.size();

                System.out.println(positive.size() + " valores positivos");
                System.out.println(dec.format(media));

            }
        }
